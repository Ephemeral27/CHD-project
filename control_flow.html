Control Flow

ðŸ”¹ if / else

Used to make decisions.


let age = 20;

if (age > 18) {
  console.log("Adult");
} else {
  console.log("Minor");
}
---
ðŸ”¹ switch

Cleaner way to check multiple cases.


let day = "Monday";

switch (day) {
  case "Monday":
    console.log("Start of week");
    break;
  case "Friday":
    console.log("Weekend soon!");
    break;
  default:
    console.log("Regular day");
}
---
ðŸ”¹ for loop

Repeat something a certain number of times.


for (let i = 0; i < 5; i++) {
  console.log(i);
}
---
ðŸ”¹ while loop

Repeat while something is true.


let i = 0;

while (i < 5) {
  console.log(i);
  i++;
}
---
ðŸ”¹ for...of

Loop through arrays.


let fruits = ["apple", "banana", "orange"];

for (let fruit of fruits) {
  console.log(fruit);
}
---
ðŸ”¹ for...in

Loop through object keys.


let person = {
  name: "Salma",
  age: 16
};

for (let key in person) {
  console.log(key, person[key]);
}

Goal: You should feel comfortable repeating code using loops.
